PhClick:	lda	mouseData
	beq	10$
	rts
10$:	mvb PhPgDims,PhClickA+1
	ldw	A9,#PhPgRefs-2
PhClickA:	lda	#$31
	cmp PhPgDims+1
	bcc	10$
	rts
10$:	sta	R2L
	sta	R2H
	avb	#$09,R2H	; fix bottom
	mvb	R2H,PhClickA+1	; bottom is next top
	inw	A9	; setup vlir ref ptr
	inw	A9
	ldy	#0	; are we pointing nowhere?
	lda	(A9),y
	beq	19$
	mvw	PhPgDims+2,R3
	mvw	PhPgDims+4,R4
	jsr	IsMseInRegion	; did we click HERE
	cmp	#$00
	beq	PhClickA	; no, so check next box
	jsr	InvertRectangle	; invert where we clicked!
	cpw	A9,PhSelRef	; was this clicked before?
	bne	20$
	ldw	PhSelRef,#0	; yes, so clear selection
19$:	rts		; exit here
20$:	ldy	#5
25$:	lda	R2,y
	sta	APITemp1,y
	lda	PhSelDims,y
	sta	R2,y
	dey
	bpl	25$
	lda	PhSelRef+1	; check PREV selection
	beq	29$	; NO prev selection. 
	jsr	InvertRectangle
29$:	ldy	#5
30$:	lda	APITemp1,y
	sta	PhSelDims,y
	dey
	bpl	30$
40$:	mvw	A9,PhSelRef
	rts

MovePhPg:	;.byte	$05,$74	; col,row
	;.byte	$0b,$10	; pic width and height
	;.byte	$1f,$a9,$36,$00,$08,$01	; dialog box dimensions
	ldb	R2L,#($1f + $74)
	ldb	R2H,#($1f + $84)
	ldw	R3,#$59
	mvw	R3,R4
	avw	#$15,R4
	jsr	IsMseInRegion	; did we click HERE
	cmp	#$00
	beq	10$	; no, so check next box
	jmp	MovePhBot
10$:	mvw	R4,R3
	avw	#$15,R4
	jsr	IsMseInRegion
	cmp	#$00
	beq	20$
	jmp	MovePhTop
20$:	mvw	R4,R3
	avw	#$15,R4
	jsr	IsMseInRegion
	cmp	#$00
	beq	30$
	jmp	MovePhUp
30$:	mvw	R4,R3
	avw	#$15,R4
	jsr	IsMseInRegion
	cmp	#$00
	beq	40$
	jmp	MovePhDn
40$:	rts



MovePhUp:	cwi	PhPagesDex,#PhPages
	bne	10$
	rts
10$:	svw	#2,PhPagesDex
	jsr	MvCurrPg
	jmp	FillPhPg

	rts

	rts

	jmp	FillPhPg

	mvw	PhPagesDex,A0
	ldy	#0
	lda	PhPage
	sta	(A0),y
	iny
	lda	PhPage+1
	sta	(A0),y
	rts

MovePhTop:	cwi	PhPagesDex,#PhPages
	bne	10$
	rts
10$:	ldw	PhPage,#0
	jmp	FillPhPg

MovePhBot:	lda	PhLastFl
	beq	10$
	rts
10$:	jsr	MovePhDn
	jmp	MovePhBot

	avb	#$07,A2H
	ldw	PhPage,#0
	jsr	NextPhNm

	rts

	lda	(R0),y

	jsr	MvPgDown
	mvb	PhPgDims,A2H
	avb	#$07,A2H

	ldy	#0

	jsr	AppendRecord
	cpx	#0
	beq	10$
	rts

	tya

	iny
	bne	20$

	ldw	R0,#PhPgBlk+2

FillPhPg:	lda	#0
	sta	PhLastFl
	jsr	SetPattern
	jsr	i_Rectangle
PhPgDims:	.byte	$31,$90
	.word $003e,$00c9
	ldw	PhSelRef,#0	; clear selection ptr
	ldy	#5
10$:	lda	windowTop,y
	sta	WinSave,y
	lda	PhPgDims,y
	sta	windowTop,y
	dey
	bpl	10$
	ldy	#31
15$:	lda	#0
	sta	PhPgRefs,y
	dey
	bpl	15$	
	mvb	PhPgDims,R1H
	avb	#$07,R1H
	ldw	A9,#PhPgRefs
20$:	mvb	R1H,R10H
	jsr	NextPhNm	; get phone number
	lda	R0H
	bne	25$	; flag as err or eof
	ldb	PhLastFl,#1
	bra	30$
25$:	mvb	R10H,R1H	; now display it!
	ldy	#0
	lda	(R0),y
	beq	20$
	lda	PhPage	; got good R0
	sta	(A9),y	; so safe page ref to list
	inw	A9
	lda	PhPage+1
	sta	(A9),y
	inw	A9
	ldw	R11,#$003f
	jsr	PutString
	avb	#$09,R1H
	cpb	R1H,PhPgDims+1
	bcc	20$
30$:
	ldy	#5
90$:	lda	WinSave,y
	sta	windowTop,y
	dey
	bpl	90$
	rts

NextPhNm:	ldw	R0,#0
	lda	PhPage
	ora	PhPage+1
	bne	10$
	jsr	FirstPhNm
	beq	09$
	ldw	PhPage,#0
09$:	rts
10$:	cpb	PhPage,curRecord
	beq	20$
	lda	PhPage+1
	pha
	jsr	ThisPhBlk
	pla
	sta	PhPage+1
20$:	lda	PhPage+1
	cmp	#170
	bcs	30$
	add	#84
	sta	PhPage+1
	jmp	PhMkR0
30$:	jsr	NextPhBlk
	rts

FirstPhNm:	ldw	PhPage,#0
	ldw	PhPagesDx,#PhPages
	ldw	PhPages,#0
NextPhBlk:	ldw	R0,#0
	inc	PhPage
ThisPhBlk:	lda	PhPage
	jsr	PointRecord
	cpx	#0
	bne	09$
	cpy	#0	; y=0 means ok, but no data
	bne	10$
08$:	mvb	curRecord,usedRecords
	dec	curRecord
09$:	dec	PhPage
	ldx	#1
	rts
10$:	jsr	LoadPhPg
	bne	NextPhBlk
	mvb	curRecord,PhPage
	ldb	PhPage+1,#2
	jsr	PhMkR0
	ldx	#0
	rts



LoadPhPg:	ldw	R2,#$00fe
	ldw	R7,#PhPgBlk
	jsr	ReadRecord
	cpx	#11	; buffer overflow is OK!
	beq	10$
	cpx	#0
	bne	10$
	cbi	PhPgBlk,#$50	; 'P'
	bne	10$
	cbi	PhPgBlk+1,#$48	; 'H'
	bne	10$
	rts
10$:	ldx	#1
	rts


PhMkR0:	ldw	R0,#PhPgBlk
	lda	R0L
	clc
	adc	PhPage+1
	sta	R0L
	lda	R0H
	adc	#0
	sta	R0H
	rts
