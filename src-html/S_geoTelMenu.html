ScreenBack:	ldb	flgDoBg,#0
	ldw	R0,#SCREEN_BASE+(16*40)
	ldw	R1,#BACK_SCR_BASE+(16*40)
	ldw	R2,#$1f3f-(16*40)
	jmp	MoveData

ScreenSafe:	lda	flgDoBg
	beq	10$
	jmp	ScreenBack
10$:	rts

SavColorScn:	ldw	R0,#COLOR_MATRIX+(40*2)	; scrn color
	ldw	R1,#ColorBBuf
	ldw	R2,#$03e8-(40*2)
	jsr	MoveData
	ldw	R1,COLOR_MATRIX+(40*2)	; scrn color
	ldw	R0,#$3e8-(40*2)
	mvb	origColor,R2L
	jsr	FillRam	; color the term screen
	rts

ResColorScn:	ldw	R1,#COLOR_MATRIX+(40*2)	; scrn color
	ldw	R0,#ColorBBuf
	ldw	R2,#$03e8-(40*2)
	jsr	MoveData
	rts

DrawMenu:	lda	BuffEnable
	beq	10$
	ldb	MnuF0_1,#$06
	ldw	MnuF0_2,#MnuTxi06
10$:	lda	#[(Menu00)	; point to first menu
	sta	$02
	lda	#](Menu00)
	sta	$03
	lda	#$00
	jmp	DoMenu	; DoMenu
MenuAct:	jmp	GotoFirstMenu	; GotoFirstMenu

BackMenu:	jsr	ScreenSafe
	jsr	SavColorScn
	jsr	keyRestore
	mvw	RecoverVector, RecoMenu+1
	ldw	RecoverVector,#RecoMenu
	rts

RecoMenu:	jsr	$8fc2
	jsr	keySetup
	jsr	ResColorScn
	mvw	RecoMenu+1,RecoverVector
	rts

PoorPETSCII:	ldx	XDnFType
	cpx	#$02
	bne	10$
	rts
10$:	cmp	#$08
	bne	12$
	lda	#$14	; 8->20
11$:	rts
12$:	cmp	#$0d
	beq	11$
	cmp	#$20
	bcs	14$
13$:	lda	#0
	rts
14$:	cmp	#$41
	bcc	11$	; 32-40 is good
	cmp	#$5b
	bcs	20$
	add	#$80
	rts
20$:	cmp	#$61
	bcc	11$
	cmp	#$7b
	bcs	13$
	sec
	sbc	#$20
	rts

20$:	ldw	A2,#XFilename

	sta	A0L
	inx
	lda	MnuBauds,x

	inx

	add	#1
	asl	a
	sta	mdmRdRate

doExit:	jsr	GotoFirstMenu
	jsr	SavColorScn
	jsr	ScreenSafe
	jsr	keyRestore
	jsr	ComOnline
	bne	50$
	ldw	A2,#ExitString
	jsr	YNBox
	lda	sysDBData
	cmp	#$04
	bne	70$
10$:	jsr	keySetup
	jmp	ResColorScn
50$:	ldw	A2,#DiscString
	jsr	YNBox
	lda	sysDBData
	cmp	#$04
	beq	10$
	jsr	modal1On
	lda	#14
	jsr	Disconn2
	jsr	modal1Off
70$:	ldw	R0,#atbCmd
	jsr	ComPutS
	jmp	CleanExit
