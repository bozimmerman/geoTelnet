ScreenBack:	ldb	flgDoBg,#0
	ldw	R0,#SCREEN_BASE+(16*40)
	ldw	R1,#BACK_SCR_BASE+(16*40)
	ldw	R2,#$1f3f-(16*40)
	jmp	MoveData

ScreenSafe:	lda	flgDoBg
	beq	10$
	jmp	ScreenBack
10$:	rts

SavColorScn:	ldw	R0,#COLOR_MATRIX+(40*2)	; scrn color
	ldw	R1,#ColorBBuf
	ldw	R2,#$03e8-(40*2)
	jsr	MoveData
	ldw	R1,COLOR_MATRIX+(40*2)	; scrn color
	ldw	R0,#$3e8-(40*2)
	mvb	origColor,R2L
	jsr	FillRam	; color the term screen
	rts

	ldw	R0,#ColorBBuf

	rts

doBaud300:	lda	#0
	jmp	doBaud
doBaud1200:	lda	#1
	jmp	doBaud
doBaud2400:	lda	#2
	jmp	doBaud
doBaud4800:	lda	#3
	jmp	doBaud
doBaud7200:	lda	#4
	jmp	doBaud
doBaud9600:	lda	#5
	jmp	doBaud
doBaud:	cmp	CLL2Baud
	bne	10$
	jmp	GotoFirstMenu
10$:	sta	A2L
	jsr	GetBaudPtr
	inw	A0
	ldy	#0
20$:	lda	(A0),y
	beq	30$
	cmp	#27
	beq	30$
	sta	BaudCmd+4,y
	iny
	bne	20$
30$:	lda	#13
	sta	BaudCmd+4,y
	jsr	keyRestore
	jsr	CLL2xxRb
	ldw	R0,#BaudCmd
	jsr	ComPutS
	ldw	R0,#BaudCmd
	jsr	PutChars
	jsr	BaudSet
	jsr	FixBaud
	jsr	CLL2xxRb
	jsr	keySetup
	jmp	GotoFirstMenu	

GetBaudPtr:	asl	a
	tax
	lda	MnuBauds,x
	sta	A0L
	inx
	lda	MnuBauds,x
	sta	A0H
	rts

;* Baud Cds:	0=300, 1=1200, 2=2400,3=4800
;*	4=7200,5=9600,6=144, 7=192
FixBaud:	ldx	#0
	ldy	#0
10$:	lda	MnuBauds,x
	sta	A0L
	inx
	lda	MnuBauds,x
	beq	20$
	sta	A0H
	inx
	lda	#27
	sta	(A0),y
	bra	10$
20$:	lda	CLL2Baud
	add	#1
	asl	a
	sta	rrate
	lda	CLL2Baud
	jsr	GetBaudPtr
	mvw	A0,Menu03b
	lda	#26
	ldy	#0
	sta	(A0),y	
	rts


BaudCmd:	.byte	13,"atb2400",13,13,0


doExit:	jsr	GotoFirstMenu
	jsr	SavColorScn
	jsr	ScreenSafe
	jsr	keyRestore
	jsr	ComOnline
	bne	50$
	ldw	A2,#ExitString
	jsr	YNBox
	lda	sysDBData
	cmp	#$04
	bne	70$
10$:	jsr	keySetup
	jmp	ResColorScn
50$:	ldw	A2,#DiscString
	jsr	YNBox
	lda	sysDBData
	cmp	#$04
	beq	10$
	jsr	modal1On
	lda	#14
	jsr	Disconn2
	jsr	modal1Off
70$:	ldw	R0,#atbCmd
	jsr	ComPutS
	jmp	CleanExit

atbCmd:	.byte	13,"atb1200",13,13,13,13,13,13,0
ExitString:	.byte	24,"Exit to deskTop?",27,0
DiscString:	.byte	24,"Disconnect and Exit?",27,0

doDisconnect:	jsr	GotoFirstMenu
	jsr	SavColorScn
	jsr	ScreenSafe
	jsr	keyRestore
	; * prompt, maybe?
	jsr	modal1On
	lda	#14
	jsr	Disconn2
	jsr	modal1Off
	jsr	keySetup
	jmp	ResColorScn



Disconn2:	sta	A4H
	jsr	ComOnline
	bne	10$
	rts
10$:	lda	A4H
	jsr	Pause
	ldw	R0,#PlusPlusPlus
	jsr	ComPutS
	lda	A4H
	jsr	Pause
	ldw	R0,#athCmd
	jsr	ComPutS
	rts

PlusPlusPlus:	.byte	"+++",0
athCmd:	.byte	"ath0",13,0

Pause:	sta	A5L
	lda	$01
	pha
	lda	#$35
	sta	$01
10$:	mvb	$dc08,A5H
20$:	cpb	A5H,$dc08
	beq	20$
	dec	A5L
	bne	10$
	pla
	sta	$01
	rts

	.byte	$01,$60,$00,$50	; MOVEPENTO Xl,Xh,Y
	.byte	$05,$00	; NEWPATTERN
	.byte	$03,$ea,$00,$6d	; RETANGLETO  Xl,Xh,Y
	.byte	$06,$7c,$00,$61,24,"Disconnecting...",27,0
	rts		; exit from screen routine
