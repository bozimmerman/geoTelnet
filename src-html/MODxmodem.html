.psect

XMO_ACK	= $06
XMO_CAN	= $18
XMO_EOT	= $04
XMO_NAK	= $15
XMO_SOH	= $01
XMO_CNAK	= $43
XMO_PAD	= $1A
XMO_OTH	= $02


;****************************************
;* XMOrcvChk		X-Modem Receive (checksum)
;* XMOrcvCrc		X-Modem Receive (CRC-16)
;* Inputs:
;*	A2	Vector - Receive a single byte in .a
;*	A3	Vector - Received a bad block
;* Outputs:
;*	.x 	0 means A-OK, or abort/error
;*
;********************************
XMOrcvChk:	ldb	XMOXType,#0
	ldb	XMOBufSize,#$84
	jmp	XMOrcvSt
XMOrcvCrc:	ldb	XMOXType,#1
	ldb	XMOBufSize,#$85
XMOrcvSt:	mvw	A2,XMOrcVec+1	; receive 1 byte in .a
	mvw	A3,XMOerVec+1	; receive error block
	ldb	XMOUpDn,#1
	jsr	XMOClear
	jsr	XMOBufFlush
	inc	XMOBlock
	lda	XMOXType
	bne	10$
	lda	#XMO_CNAK
	jsr	ComSPut
10$:	jmp	XMOrcvLp

XMORcVec:	jmp	XMONada	; receive 1 byte
XMOErVec:	jmp	XMONada	; error block receive bytes
XMONada:	rts	

XMOrcvLp:	lda	XMOErrors
	cmp	#$0e
	bcc	11$
10$:	ldb	XMOAbort,#$01
11$:	jsr	XMOChkAbort
	jsr	XMOChkTm
	lda	XMOAbort
	beq	15$
	jmp	XMOXAbort
14$:	jmp	XMOXRcv
15$:	cbi	XMOBlock,#$01
	bne	17$
	cbi	XMOTmOuts,#$02
	bne	17$
	beq	10$	; to cancel
17$:	lda	XMOClock
	beq	18$
	jmp	XMONoBort
18$:	lda	XMOBuf
	cmp	#XMO_CAN
	beq	10$
	cmp	#XMO_EOT
	beq	14$
	jsr	XMOgetKey
	lda	XMOKeySync
	beq	XMOrcvLp
	jsr	XMOReTm
	ldy	XMOBufBit
	lda	XMOKeyFree
	sta	XMOBuf,y
	inc	XMOBufBit
	iny
	cpy	XMOBufSize
	bcc	XMOrcvLp	; 
	jsr	XMOPause	; Check the Block
	lda	XMOBuf
	cmp	#XMO_EOT
	beq	14$
	cmp	#XMO_CAN
	beq	10$
	cmp	#XMO_OTH
	beq	20$
	cmp	#XMO_SOH
	beq	20$
19$:	jmp	XMOConRcv
20$:	cpb	XMOBuf+1, XMOBlock
	bne	19$
	eor	#$ff
	cmp	XMOBuf+1
	bne	19$
	jsr	XMOchkBlk
	lda	XMOXType
	bne	22$
	cpb	XMOChkSum, XMOBufEnd
	beq	25$
21$:	jmp	XMOConRcv
22$:	cpb	XMOBufEnd, XMOCrc
	bne	21$
	cpb	XMOBufCrc,XMOCrc+1
	bne	21$
25$:	ldb	XMOGood,#1
	jmp	XMOConRcv

XMOConRcv:	lda	XMOGood
	beq	20$
	ldb	XMOBufCt,#0
10$:	lda	XMOBufCt,y
	jsr	XMORcVec
	inc	XMOBufCt
	cbi	XMOBufCt,#$80
	bcc	10$
	jsr	XMOBufFlush
	lda	#XMO_ACK
	jsr	ComSPut
	inc	XMOBlock
	jmp	XMOrcvLp
20$:	lda	XMOAbort
	beq	XMONoBort
	jmp	XMOXAbort
XMONoBort:	jsr	XMOBufFlush
	jsr	XMOerVec
	lda	XMOXType
	bne	30$
25$:	lda	#XMO_NAK
	jsr	ComSPut
	bra	40$
30$:	lda	XMOBlock
	cmp	#$02
	bcs	25$
	lda	#XMO_CNAK
	jsr	ComSPut
40$:	inc	XMOErrors
	jmp	XMOrcvLp

XMOXRcv:	lda	#XMO_ACK
	jsr	ComSPut
	jmp	XMOExit

XMOXAbort:	lda	#$01
	sta	XMOAbort
	lda	#XMO_CAN
	jsr	ComSPut
	lda	#XMO_CAN
	jsr	ComSPut
	lda	#XMO_CAN
	jsr	ComSPut
	jmp	XMOExit

XMOExit:	ldx	XMOAbort
	jsr	XMOClear
	stx	XMOAbort
	jsr	XMOBufFlush
	lda	#0
	sta	XMOXType
	sta	XMOUpDn
	lda	XMOAbort
	tax
	rts


;****************************************
;* XMOsend		X-Modem Send
;* Inputs:
;*	A2	Vector - Ensure 128 bytes available
;*	A3	Vector - Send a single byte in .a, .x=no more
;*	A4	Vector - Sent a bad block
;* Outputs:
;*	.x 	0 means A-OK, or abort/error
;*
;********************************

XMOsend:	ldb	XMOUpDn,#0
	jsr	XMOClear
	jsr	XMOBufFlush
	jsr	XMOgosCrc
	mvw	A2,XMOSsVec+1
	mvw	A3,XMOSnVec+1
	mvw	A4,XMOErVec+1
	jmp	XMOsendStart

XMOSsVec:	jmp	XMONada	; prepare 128 bytes
XMOSnVec:	jmp	XMONada	; send 1 byte

XMOgosChk:	ldb	XMOXType,#$00
	ldb	XMOBufSize,#$84
	rts
XMOgosCrc:	ldb	XMOXType,#$01
	ldb	XMOBufSize,#$85
	rts



XMObuild:	jsr	XMOSsVec
	ldy	#0
	sty	XMOIOStat
	sty	XMOBufCt
10$:	jsr	XMOSnVec
	ldy	XMOBufCt
	sta	XMOBuf3,y
	stx	XMOIOStat
	cpx	#0
	bne	20$
	inc	XMOBufCt
	cbi	XMOBufCt,#$80
	bcc	10$
	bra	30$
20$:	inc	XMOBufCt
	inc	XMOBufCt
25$:	ldy	XMOBufCt
	lda	#XMO_PAD
	sta	XMOBuf3,y
	inc	XMOBufCt
	cbi	XMOBufCt,#$80
	bcc	25$
30$:	ldb	XMOBuf,#XMO_SOH
	lda	XMOBlock
	sta	XMOBuf+1
	eor	#$ff
	sta	XMOBuf+2
	jsr	XMOchkBlk
	lda	XMOChkSum
	sta	XMOBufEnd
	lda	XMOXType
	beq	XMOcontSend
	cpb	XMOCrc,XMOBufEnd
	cpb	XMOCrc+1,XMOBufCrc
XMOcontSend:	jsr	XMOminFlush
	ldy	#0
	sta	XMOBufBit
50$:	lda	XMOBuf,y
	jsr	ComSPut
	ldy	XMOBufBit
	iny
	sty	XMOBufBit
	cpy	XMOBufSize
	bcc	50$
	jmp	XMOsendStart

XMOsendStart:	jsr	XMOReTm
10$:	jsr	XMOChkAbort
	jsr	XMOChkTm
	jsr	ComGet
	bcs	19$
	cmp	#XMO_NAK
	beq	33$
	cmp	#XMO_CNAK
	beq	32$
	cmp	#XMO_CAN
	beq	30$
	cmp	#XMO_ACK
	beq	50$
19$:	cbi	XMOErrors,#$0e
	beq	30$
	lda	XMOClock
	bne	40$
	bra	10$
30$:	jmp	XMOXAbort
32$:	jsr	XMOgosCrc
	bra	35$
33$:	jsr	XMOgosChk
35$:	lda	XMOBlock
	beq	50$
40$:	inc	XMOErrors
	jsr	XMOErVec
	jsr	XMOPause
	jmp	XMOcontSend
50$:	inc	XMOBlock
	lda	XMOIOStat
	beq	80$
	ldy	#0
70$:	tya
	pha
	lda	#XMO_EOT
	jsr	ComSPut
	pla
	tay
	iny
	cpy	#$85
	bne	70$
	jmp	XMOExit
80$:	jsr	XMOPause
	jsr	XMOBufFlush
	jmp	XMObuild	

XMOClear:	lda	#0
	sta	XMOBlock
	sta	XMOGood
	sta	XMOErrors
	sta	XMOAbort
	sta	XMOBufBit
	sta	XMOChkSum
	sta	XMOCrc
	sta	XMOCrc+1
	sta	XMOTmOuts
	sta	XMOIOStat
	rts



XMOReTm:	lda	$01
	pha
	lda	#$35
	sta	$01
	lda	#0
	sta	XMOChkC
	sta	$dc09
	sta	$dc08
	pla
	sta	$01
	rts

XMOChkTm:	ldb	XMOClock,#0
	lda	$01
	pha
	lda	#$35
	sta	$01
	lda	$dc09
	beq	10$
	inc	XMOChkC
	lda	#0
	sta	$dc09
	sta	$dc08
10$:	lda	$dc08
	pla
	sta	$01
	cbi	XMOClock,#3
	bcs	XMOBadTm
	rts
XMOBadTm:	inc	XMOTmOuts
	inc	XMOClock
	jmp	XMOReTm



XMOChkAbort:	; TODO: * Is C= key held down?
	jsr	ComOnline
	beq	10$
	lda	#0
	rts
10$:	ldb	XMOAbort,#1
	rts

XMOErrFlush:	inc	XMOErrors
XMOBufFlush:	ldy	#0
	tya
10$:	sta	XMOBuf,y
	iny
	cpy	#$85
	bcc	10$
XMOMinFlush:	jsr	CLL2xxRb
	lda	#0
	sta	XMOBufBit
	sta	XMOGood
	jsr	XMOReTm
	rts

XMOPause:	lda	$01
	pha
	lda	#$35
	sta	$01
	lda	#0
	sta	APITemp2
10$:	lda	#0
	sta	$dc08
20$:	lda	$dc08
	beq	20$
	inc	APITemp2
	cbi	APITemp2,#9	; almost a second
	bcc	10$
	pla
	sta	$01
	rts


XMOgetKey:	tya
	pha
	lda	#0
	sta	XMOKeySync
	jsr	ComGet
	bcc	10$
	pla
	tay
	rts
10$:	sta	XMOKeyFree
	inc	XMOKeySync
	pla
	tay
	rts

XMOchkBlk:	lda	XMOXType
	bne	10$
	ldy	#0
	sta	XMOChkSum
10$:	lda	XMOBuf3,y
	add	XMOChkSum
	sta	XMOChkSum
	cpy	#$80
	bcc	10$
	rts

XMOchkCrc:	ldy	#0
	sty	XMOCrc
	sty	XMOCrc+1
10$:	lda	XMOBuf3,y
	eor	XMOCrc
	tax
	lda	XMOcrcMB,x
	eor	XMOCrc+1
	sta	XMOCrc
	lda	XMOcrcLB,x
	sta	XMOCrc+1
	iny
	cpy	#$80
	bne	10$
	rts
