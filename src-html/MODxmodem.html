XMOClear:	lda	#0
	sta	XMOBlock
	sta	XMOGood
	sta	XMOErrors
	sta	XMOAbort
	sta	XMOBufBit
	sta	XMOChkSum
	sta	XMOCrc
	sta	XMOCrc+1
	sta	XMOTmOuts
	sta	XMOIOStat
	rts



XMOReTm:	lda	$01
	pha
	lda	#$35
	sta	$01
	lda	#0
	sta	XMOChkC
	sta	$dc09
	sta	$dc08
	pla
	sta	$01
	rts

XMOChkTm:	ldb	XMOClock,#0
	lda	$01
	pha
	lda	#$35
	sta	$01
	lda	$dc09
	beq	10$
	inc	XMOChkC
	lda	#0
	sta	$dc09
	sta	$dc08
10$:	lda	$dc08
	pla
	sta	$01
	cbi	XMOClock,#3
	bcs	XMOBadTm
	rts
XMOBadTm:	inc	XMOTmOuts
	inc	XMOClock
	jmp	XMOReTm



XMOChkAbort:	; TODO: * Is C= hty held down?
	jsr	IsConnected
	beq	10$
	lda	#0
	rts
10$:	ldb	XMOAbort,#1
	rts

XMOErrFlush:	inc	XMOErrors
XMOBufFlush:	ldy	#0
	tya
10$:	sta	XMOBuf,y
	iny
	cpy	#$85
	bcc	10$
XMOMinFlush:	jsr	CLL2xxRb
	lda	#0
	sta	XMOBufBit
	sta	XMOGood
	jsr	XMOReTm
	sta
